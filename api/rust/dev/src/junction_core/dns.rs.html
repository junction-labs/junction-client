32/// An error that occurred while parsing a system DNS configuration.
33#[derive(Debug, thiserror::Error)]
34pub(crate) enum ConfigError {
35    #[error("config not found: {path}")]
36    NotFound { path: String },
37
38    #[error("{path}:{line}: {message}")]
39    Invalid {
40        path: String,
41        line: usize,
42        message: String,
43    },
44
45    #[error(transparent)]
46    Other(#[from] std::io::Error),

56/// Load a [SystemConfig] from the given path. You probaly want to read
57/// `/etc/resolv.conf` but the option is here just in case you don't.
58pub(crate) fn load_config<P: AsRef<Path>>(path: P) -> Result<SystemConfig, ConfigError> {
59    let content = match std::fs::read(path.as_ref()) {
60        Ok(content) => content,
61        Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
62            return Err(ConfigError::NotFound {
63                path: path.as_ref().display().to_string(),
64            })
65        }
66        Err(e) => return Err(e.into()),
67    };
68
69    parse_resolv_conf(path, &content)
70}

72/// An extremely simple resolv.conf parser.
73///
74/// This is included here so we don't have to take another dependency - the
75/// complexity of parsing two options out of a text file felt quite low, and as
76/// of writing, the crate(s) available for it didn't appear to be doing anything
77/// more sophisticated than this or have extensive fuzz testing regimes, etc.
78fn parse_resolv_conf(path: impl AsRef<Path>, content: &[u8]) -> Result<SystemConfig, ConfigError> {
79    // split on newline
80    let lines = content.split(|&b| b == b'\n');
81
82    let mut ndots = 1u8;
83    let mut search = vec![];
84
85    for (i, line) in lines.enumerate() {
86        let line = line.trim_ascii();
87
88        match line.first() {
89            // skip comments
90            Some(b';') | Some(b'#') => continue,
91            // skip empty lines
92            None => continue,
93            _ => (),
94        }
95
96        let parts: Vec<_> = line.split(|b| b.is_ascii_whitespace()).collect();
97        match parts.as_slice() {
98            [b"options", options @ ..] => {
99                for option in options {
100                    if let Some(n) = option.strip_prefix(b"ndots:") {
101                        ndots = parse_as_str(n).map_err(|()| ConfigError::Invalid {
102                            path: path.as_ref().display().to_string(),
103                            line: i,
104                            message: format!("invalid ndots: '{}'", String::from_utf8_lossy(n)),
105                        })?;
106                    }
107                }
108            }
109            [b"search", hostnames @ ..] => {
110                let hostnames: Result<Vec<_>, _> =
111                    hostnames.iter().map(|bs| Hostname::try_from(*bs)).collect();
112
113                match hostnames {
114                    Ok(hostnames) => search = hostnames,
115                    Err(e) => {
116                        return Err(ConfigError::Invalid {
117                            path: path.as_ref().display().to_string(),
118                            line: i,
119                            message: format!("search path contains invalid hostname: {e}"),
120                        })
121                    }
122                }
123            }
124            // ignore any other directives, even if they're badly formed. we
125            // don't care about em
126            _ => (),
127        }
128    }
129
130    Ok(SystemConfig { ndots, search })
131}

133fn parse_as_str<T: std::str::FromStr>(bs: &[u8]) -> Result<T, ()> {
134    let as_str = std::str::from_utf8(bs).map_err(|_| ())?;
135    as_str.parse().map_err(|_| ())
136}

138/// A blocking resolver that uses the stdlib to resolve hostnames to addresses.
139///
140/// Names are resolved regularly in the background. If the addresses behind a
141/// name change, or a resolution error occurs, the results are broadcast over
142/// a channel (see the `subscribe` method) to all subscribers.
143///
144/// Behind the scenes, this resolver uses a fixed pool of threads to
145/// periodically resolve all of the addresses for a name. On every resolution,
146/// the returned set of IP addresses is treated as the entire set of addresses
147/// that make up a name and overwrites any previous addreses.  This roughly
148/// corresponds to Envoy's [STRICT_DNS] approach to resolution.
149///
150/// A StdlibResolver spawns its own pool of worker threads in the background
151/// that exit when the resolver is dropped.
152///
153/// [STRICT_DNS]: https://www.envoyproxy.io/docs/envoy/latest/intro/arch_overview/upstream/service_discovery#strict-dns
154#[derive(Clone, Debug)]
155pub(crate) struct StdlibResolver {
156    inner: Arc<StdlibResolverInner>,
157}
158
159#[derive(Debug)]
160struct StdlibResolverInner {
161    lookup_interval: Duration,
162    lookup_jitter: Duration,
163
164    // a mutex/condvar pair wrapped around resolver state. see ResolverState for
165    // exactly what that does. nothing here should be much more than locking
166    // around accessing that struct.
167    cond: Condvar,
168    tasks: Mutex<ResolverState>,
169
170    // a notify to let async callers know names have changed while they're
171    // waiting. not specific to any name - will definitely get busy and have
172    // spurious wakeups at hgih volumes of names.
173    async_notify: Notify,
174
175    // the the number of worker threads the resolver was started with. if the
176    // number of references to this struct ever drops, it's time for the
177    // resolver threads to shut down.
178    worker_count: usize,
179}
180
181macro_rules! no_poison {
182    ($guard:expr) => {
183        $guard.expect("SystemResolver was poisoned: this is a bug in Junction")
184    };
185}

187impl Drop for StdlibResolver {
188    fn drop(&mut self) {
189        // self.shutdown();
190    }

193impl StdlibResolver {
194    pub(crate) fn new_with(
195        lookup_interval: Duration,
196        lookup_jitter: Duration,
197        threads: usize,
198    ) -> Self {
199        let inner = StdlibResolverInner {
200            lookup_interval,
201            lookup_jitter,
202            tasks: Mutex::new(ResolverState::default()),
203            cond: Condvar::new(),
204            async_notify: Notify::new(),
205            worker_count: threads,
206        };
207        let resolver = StdlibResolver {
208            inner: Arc::new(inner),
209        };
210
211        for _ in 0..threads {
212            let resolver = resolver.clone();
213            std::thread::spawn(move || resolver.run());
214        }
215
216        resolver
217    }

219    pub(crate) fn get_endpoints(
220        &self,
221        hostname: &Hostname,
222        port: u16,
223    ) -> Option<Arc<EndpointGroup>> {
224        let tasks = no_poison!(self.inner.tasks.lock());
225        tasks.get_endpoints(hostname, port)
226    }
227
228    pub(crate) async fn get_endpoints_await(
229        &self,
230        hostname: &Hostname,
231        port: u16,
232    ) -> Option<Arc<EndpointGroup>> {
233        // fast path: the endpoints are in the map.
234        if let Some(endpoints) = self.get_endpoints(hostname, port) {
235            return Some(endpoints);
236        }
237
238        // slow path: we're waiting
239        //
240        // on every run through the loop, we need to, IN ORDER, do the
241        // following:
242        //
243        // - register for notifications
244        // - try to get a name in the map, returning it if present
245        // - wait for the next notification
246        //
247        // given this order of events, and the guarantees on Notify, there is no
248        // interleaving of events where we can miss a notification.
249        //
250        // SAFETY: this assumes that the writer half of a notify is always using
251        // notify_waiters and not ever using notify_one.
252        //
253        // for an example that uses notify_one() instead, see Notify
254        //
255        // https://docs.rs/tokio/latest/tokio/sync/futures/struct.Notified.html#method.enable
256        let changed = self.inner.async_notify.notified();
257        tokio::pin!(changed);
258        loop {
259            // check the map
260            if let Some(entry) = self.get_endpoints(hostname, port) {
261                return Some(entry);
262            }
263
264            // wait for a change
265            changed.as_mut().await;
266
267            // this uses Pin::set so we're not allocating/deallocating a new
268            // wakeup future every time.
269            changed.set(self.inner.async_notify.notified());
270        }
271    }

273    pub(crate) fn subscribe(&self, name: Hostname, port: u16) {
274        let mut tasks = no_poison!(self.inner.tasks.lock());
275
276        // on a new subscribtion, notify both the background workers and any
277        // async tasks waiting on a get.
278        if tasks.pin(name, port) {
279            self.inner.cond.notify_all();
280            self.inner.async_notify.notify_waiters();
281        }
282    }
283
284    pub(crate) fn unsubscribe(&self, name: &Hostname, port: u16) {
285        let mut tasks = no_poison!(self.inner.tasks.lock());
286        tasks.remove(name, port);
287        self.inner.cond.notify_all();
288    }

290    pub(crate) fn set_names(&self, new_names: impl IntoIterator<Item = (Hostname, u16)>) {
291        let new_names = new_names.into_iter();
292
293        let mut tasks = no_poison!(self.inner.tasks.lock());
294        if tasks.update_all(new_names) {
295            self.inner.cond.notify_all();
296        }
297    }

299    pub(crate) fn run(&self) {
300        let thread_id = std::thread::current().id();
301        macro_rules! worker_trace {
302            ($($item:tt)*) => {
303                tracing::trace!(
304                    ?thread_id,
305                    worker_count = self.inner.worker_count,
306                    strong_count = Arc::strong_count(&self.inner),
307                    $(
308                        $item
309                    )*
310                )
311            };
312        }
313
314        worker_trace!("starting");
315        loop {
316            // grab the next name
317            let Some(name) = self.next_name() else {
318                worker_trace!("exiting");
319                return;
320            };
321
322            // do the DNS lookup
323            //
324            // this always uses 80 and then immediately discards the port. we
325            // don't actually care about what the port is here.
326            worker_trace!(%name, "starting lookup");
327            let addr = (&name[..], 80);
328            let answer = std::net::ToSocketAddrs::to_socket_addrs(&addr).map(|answer| {
329                // TODO: we're filtering out every v6 address here. this isn't
330                // corect long term - we need to define how we want to control
331                // v4/v6 at the api level.
332                answer.filter(|a| a.is_ipv4()).collect()
333            });
334
335            // save the answer
336            worker_trace!(%name, "resolved");
337            self.insert_answer(name, Instant::now(), answer);
338        }
339    }
340
341    fn is_shutdown(&self) -> bool {
342        Arc::strong_count(&self.inner) <= self.inner.worker_count
343    }
344
345    fn next_name(&self) -> Option<Hostname> {
346        let mut tasks = no_poison!(self.inner.tasks.lock());
347
348        loop {
349            if self.is_shutdown() {
350                return None;
351            }
352
353            // claim a name older than the cutoff
354            let before = Instant::now() - self.inner.lookup_interval;
355            if let Some(name) = tasks.next_name(before) {
356                return Some(name.clone());
357            }
358
359            // if there's nothing to do, sleep until there is. add a little
360            // bit of jitter here to spread out the load this puts on upstream
361            // dns servers.
362            //
363            // if there's no task, just sleep until notified
364            let wait_time = tasks.min_resolved_at().map(|t| {
365                let d = t.saturating_duration_since(Instant::now());
366                d + self.inner.lookup_interval + rng_jitter(self.inner.lookup_jitter)
367            });
368
369            tracing::trace!(?wait_time, "waiting for new name");
370            match wait_time {
371                Some(wait_time) => {
372                    (tasks, _) = no_poison!(self.inner.cond.wait_timeout(tasks, wait_time));
373                }
374                None => tasks = no_poison!(self.inner.cond.wait(tasks)),
375            }
376        }
377    }
378
379    fn insert_answer(
380        &self,
381        name: Hostname,
382        resolved_at: Instant,
383        answer: io::Result<Vec<SocketAddr>>,
384    ) {
385        // grab the lock in a tight scope
386        {
387            let mut tasks = no_poison!(self.inner.tasks.lock());
388            tasks.insert_answer(&name, resolved_at, answer);
389        }
390
391        // after the lock is free:
392        //
393        // notify all async callers in get_await. since there's no
394        // blocking get_* method, there are no sync callers to notify.
395        self.inner.async_notify.notify_waiters();
396    }
397}

399fn rng_jitter(max: Duration) -> Duration {
400    let secs = crate::rand::with_thread_rng(|rng| rng.gen_range(0.0..max.as_secs_f64()));
401
402    Duration::from_secs_f64(secs)
403}

408#[derive(Debug, Default)]
409struct NameInfo {
410    ports: BTreeMap<u16, PortInfo>,
411    in_flight: bool,
412    resolved_at: Option<Instant>,
413    last_addrs: Option<Vec<SocketAddr>>,
414    last_error: Option<io::Error>,
415}
416
417#[derive(Debug, Default)]
418struct PortInfo {
419    pinned: bool,
420    endpoint_group: Option<Arc<EndpointGroup>>,
421}
422
423impl PortInfo {
424    fn set_addrs(&mut self, port: u16, addrs: &[SocketAddr]) {
425        let addrs = addrs.iter().cloned().map(|mut addr| {
426            addr.set_port(port);
427            addr
428        });
429        self.endpoint_group = Some(Arc::new(EndpointGroup::from_dns_addrs(addrs)))
430    }
431}
432
433impl NameInfo {
434    fn merge_answer(&mut self, now: Instant, answer: io::Result<Vec<SocketAddr>>) {
435        // always update time
436        self.resolved_at = Some(now);
437
438        // update eitehr the endpoints or error based on the answer
439        match answer {
440            Ok(mut addrs) => {
441                self.last_error = None;
442
443                // normalize addrs
444                addrs.sort();
445
446                // if the addrs have changed, update both the raw addrs and the
447                // EndpointGroup for each port.
448                if Some(&addrs) != self.last_addrs.as_ref() {
449                    for (port, port_info) in self.ports.iter_mut() {
450                        port_info.set_addrs(*port, &addrs);
451                    }
452                    self.last_addrs = Some(addrs);
453                }
454            }
455            Err(e) => self.last_error = Some(e),
456        }
457    }
458
459    fn resolved_before(&self, t: Instant) -> bool {
460        match self.resolved_at {
461            Some(resolved_at) => resolved_at < t,
462            None => true,
463        }
464    }
465}

467impl ResolverState {
468    fn next_name(&mut self, before: Instant) -> Option<&Hostname> {
469        let mut min: Option<(_, &mut NameInfo)> = None;
470
471        for (name, state) in &mut self.0 {
472            if state.in_flight {
473                continue;
474            }
475
476            match state.resolved_at {
477                Some(t) => {
478                    if t <= before && min.as_ref().map_or(true, |(_, s)| s.resolved_before(t)) {
479                        min = Some((name, state))
480                    }
481                }
482                None => {
483                    state.in_flight = true;
484                    return Some(name);
485                }
486            }
487        }
488
489        min.map(|(name, state)| {
490            state.in_flight = true;
491            name
492        })
493    }
494
495    fn min_resolved_at(&self) -> Option<Instant> {
496        self.0.values().filter_map(|state| state.resolved_at).min()
497    }
498
499    fn get_endpoints(&self, hostname: &Hostname, port: u16) -> Option<Arc<EndpointGroup>> {
500        let name_info = self.0.get(hostname)?;
501        let port_info = name_info.ports.get(&port)?;
502        port_info.endpoint_group.clone()
503    }
504
505    fn insert_answer(
506        &mut self,
507        hostname: &Hostname,
508        resolved_at: Instant,
509        answer: io::Result<Vec<SocketAddr>>,
510    ) {
511        // only update if there's still state for this name.
512        //
513        // if there's no state for this name it's because the set of target
514        // names changed and we're not interested anymore.
515        if let Some(state) = self.0.get_mut(hostname) {
516            state.in_flight = false;
517            state.merge_answer(resolved_at, answer);
518        }
519    }
520
521    fn pin(&mut self, hostname: Hostname, port: u16) -> bool {
522        let (mut created, name_info) = match self.0.entry(hostname) {
523            btree_map::Entry::Vacant(entry) => (true, entry.insert(Default::default())),
524            btree_map::Entry::Occupied(entry) => (false, entry.into_mut()),
525        };
526
527        let (port_created, port_info) = match name_info.ports.entry(port) {
528            btree_map::Entry::Vacant(entry) => (true, entry.insert(Default::default())),
529            btree_map::Entry::Occupied(entry) => (false, entry.into_mut()),
530        };
531        created |= port_created;
532        port_info.pinned = true;
533
534        if let Some(addrs) = &name_info.last_addrs {
535            port_info.set_addrs(port, addrs);
536        }
537
538        created
539    }
540
541    fn remove(&mut self, hostname: &Hostname, port: u16) {
542        let mut remove = false;
543        if let Some(entry) = self.0.get_mut(hostname) {
544            entry.ports.remove(&port);
545            remove = entry.ports.is_empty();
546        };
547
548        if remove {
549            self.0.remove(hostname);
550        }
551    }
552
553    fn update_all(&mut self, new_names: impl IntoIterator<Item = (Hostname, u16)>) -> bool {
554        // build an index of name -> [port] for the union of all names in the
555        // new set of names and the old set of names.
556        //
557        // this currently involves recloning every key in this map, but that
558        // should be ok since we expect hostname clones to be (relatively)
559        // cheap.
560        let mut names: BTreeMap<_, Vec<_>> = BTreeMap::new();
561        for name in self.0.keys() {
562            names.insert(name.clone(), Vec::new());
563        }
564        for (name, port) in new_names {
565            names.entry(name).or_default().push(port);
566        }
567
568        // iterate through the names index, for every set of ports, modify the
569        // name info so it only contains the listed ports or any existing pinned
570        // ports. the APIs for removing an entry we've already creatd here are not
571        // good, so don't actually do removal in this step.
572        let mut changed = false;
573        for (name, new_ports) in &names {
574            let name_info = self.0.entry(name.clone()).or_default();
575
576            let mut to_remove = BTreeSet::new();
577            for (port, port_info) in &name_info.ports {
578                if port_info.pinned {
579                    continue;
580                }
581                to_remove.insert(*port);
582            }
583
584            for port in new_ports {
585                to_remove.remove(port);
586                if let btree_map::Entry::Vacant(e) = name_info.ports.entry(*port) {
587                    changed = true;
588                    e.insert(PortInfo::default());
589                }
590            }
591
592            for port in to_remove {
593                changed |= name_info.ports.remove(&port).is_some();
594            }
595        }
596
597        // take another pass through to remove every (k, v) pair where v
598        // doesn't actually have any ports to keep track of.
599        self.0.retain(|_, info| !info.ports.is_empty());
600
601        changed
602    }

769        assert!(resolver.next_name(now).is_some());

772        assert_eq!(
773            resolver.names_and_ports(),
774            &[
775                ("doesnotexistihopereallybad.com", vec![80]),

778        );

780        // resolve one name.

783            now,
784            Ok(vec![SocketAddr::new(IpAddr::V4(Ipv4Addr::LOCALHOST), 80)]),

787        assert!(resolver.next_name(now - Duration::from_millis(1)).is_none());
